on_action = {
	on_weekly = {
		effect = {
			set_variable = {
				mil_factories_income = num_of_military_factories
			}
			multiply_variable = {
				mil_factories_income = 30
			}
			add_to_variable = {
				cash = mil_factories_income
			}

			set_variable = {
				doc_factories_income = num_of_naval_factories
			}
			multiply_variable = {
				doc_factories_income = 50
			}
			add_to_variable = {
				cash = doc_factories_income
			}

			set_variable = {
				civ_factories_income = num_of_civilian_factories
			}
			multiply_variable = {
				civ_factories_income = 100
			}
			add_to_variable = {
				cash = civ_factories_income
			}

			set_variable = {
				new_var = ROOT.max_manpower_k
			}
			multiply_variable = {
				max_manpower_k = 0.01
			}
			set_variable = {
				healthcare_expense = ROOT.max_manpower_k
			}
			if = {
				limit = {
					has_idea = universal_health_care
				}
				multiply_variable = {
					healthcare_expense = 6.5
				}
				else_if = {
					limit = {
						has_idea = free_checkups
					}
					multiply_variable = {
						healthcare_expense = 4.5
					}
				}
				else_if = {
					limit = {
						has_idea = free_emergencies_only
					}
					multiply_variable = {
						healthcare_expense = 1.85
					}
				}
				else_if = {
					limit = {
						has_idea = private_healthcare
					}
					multiply_variable = {
						healthcare_expense = 0.0
					}
				}
			}
			set_variable = {
				healthcare_expense_universal_health_care = healthcare_expense
			}
			multiply_variable = {
				healthcare_expense_universal_health_care = 6.5
			}
			set_variable = {
				healthcare_expense_free_checkups = healthcare_expense
			}
			multiply_variable = {
				healthcare_expense_free_checkups = 4.5
			}
			set_variable = {
				healthcare_expense_free_emergencies_only = healthcare_expense
			}
			multiply_variable = {
				healthcare_expense_free_emergencies_only = 1.85
			}
			set_variable = {
				healthcare_expense_private_healthcare = healthcare_expense
			}
			multiply_variable = {
				healthcare_expense_private_healthcare = 0.0
			}
		}
	}
}
